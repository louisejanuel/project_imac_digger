project(IMAC_project_digger)

set(CMAKE_CXX_STANDARD 17)

# Inclure les headers GLFW

# Lier le dossier contenant glfw3.lib

# Executable
add_executable(digger
    src/main.cpp
    src/MapGenerator.cpp
    src/Renderer.cpp
)

target_link_libraries(digger glfw3 opengl32)
